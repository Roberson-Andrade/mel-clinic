CREATE TABLE especie (id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY, nome VARCHAR(100) NOT NULL);
CREATE TABLE profissional (id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY, codigo INT(11) NOT NULL, nome VARCHAR(250) NOT NULL, telefone VARCHAR(50), especialidade VARCHAR(100));
CREATE TABLE procedimento (id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY, codigo INT(11) NOT NULL, descricao VARCHAR(250), valor DECIMAL(9, 2) NOT NULL, numSessao INT(11) NOT NULL, orientacoes VARCHAR(250));
CREATE TABLE pessoa (id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY, nome varchar(250) NOT NULL, endereco varchar(500), cep INT(11), cidade VARCHAR(100), uf VARCHAR(2), foneContato VARCHAR(100));
CREATE TABLE animal (id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY, proprietarioId INT(11) NOT NULL, nome VARCHAR(250) NOT NULL, raca VARCHAR(50), sexo VARCHAR(10), nascimento DATE, FOREIGN KEY (proprietarioId) REFERENCES pessoa(id) ON DELETE CASCADE);
CREATE TABLE agendamento (id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY, animalId INT(11) NOT NULL, profissionalId INT(11) NOT NULL, procedimentoId INT(11) NOT NULL, dataAgendamento DATE NOT NULL, numSessao INT(11), valorCobrado DECIMAL(9,2) NOT NULL, pagamento ENUM("CARTAO_CREDITO", "DINHEIRO", "BOLETO", "DEPOSITO", "CONVENIO"), FOREIGN KEY (animalId) REFERENCES animal(id) ON DELETE CASCADE, FOREIGN KEY (profissionalId) REFERENCES profissional(id) ON DELETE CASCADE, FOREIGN KEY (procedimentoId) REFERENCES procedimento(id) ON DELETE CASCADE);
CREATE INDEX dateAgendamentoIndex ON agendamento (dataAgendamento);
